XOR two iterators until one of them exhausts.

Example

use rustopals::util::iter::Xorable;

const EXPECTED_DISTANCE: u32 = 37;

const A: &[u8] = &[0u8, 1, 2, 3];
const B: &[u8] = &[1u8, 3, 3, 7];
const EXPECTED: &[u8] = &[1u8, 2, 1, 4];

let result = A
    .iter()
    .xor(B.iter())
    .collect::<Vec<_>>();

assert_eq!(result, EXPECTED);
